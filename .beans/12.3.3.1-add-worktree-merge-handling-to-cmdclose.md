### Changes to make

1. **Add import** at top of file:
   ```rust
   use crate::worktree::{detect_worktree, commit_worktree_changes, merge_to_main, cleanup_worktree, MergeResult};
   ```

2. **Add worktree handling after verify passes** (after line ~235, where "Verify passed" is printed):
   ```rust
   // After verify passes, handle worktree merge
   if let Some(worktree_info) = detect_worktree()? {
       // Commit any uncommitted changes
       commit_worktree_changes(&format!("Close bean {}: {}", id, bean.title))?;
       
       // Merge to main
       match merge_to_main(&worktree_info, id)? {
           MergeResult::Success | MergeResult::NothingToCommit => {
               // Continue to archive
           }
           MergeResult::Conflict { files } => {
               eprintln!("Merge conflict in files: {:?}", files);
               eprintln!("Resolve conflicts and run `bn close {}` again", id);
               return Ok(()); // Don't archive yet
           }
       }
   }
   ```

3. **Add worktree cleanup after archiving** (after bean is archived, before "Closed bean" print):
   ```rust
   // Clean up worktree after successful close
   // Note: We need to detect again as we're still in the worktree
   if let Some(worktree_info) = detect_worktree()? {
       if let Err(e) = cleanup_worktree(&worktree_info) {
           eprintln!("Warning: Failed to cleanup worktree: {}", e);
           // Don't fail the close for cleanup issues
       }
   }
   ```
